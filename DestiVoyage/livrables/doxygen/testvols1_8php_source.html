41 ),
42 );
43
44 $context = stream_context_create($options);
45 $response = file_get_contents($apiUrl, false, $context);
46
47 if ($response !== false) {
48 $data = json_decode($response, true);
49 if (isset($data['access_token'])) {

51 $_SESSION['lastTokenTime'] = time();

53 }
54 }
